sudo apt install inotify-tools

sudo inotifywait -m -r /home


sudo inotifywait -m -r -e access,modify,create,delete ~/

-m: Monitor the directory continuously.
-r: Recursively monitor all subdirectories.
-e: Specify the events to watch for. In this case, we are watching for:
access: Read operations.
modify: Modifications to files.
create: Creation of new files.
delete: Deletion of files.


find the last 100 files that have been written to 
with unix timestamp 
sudo find / -xdev \( -path /proc -o -path /sys -o -path /dev \) -prune -o -type f -printf '%T@ %p\n' | sort -n | tail -n 100

send error messages (stderr) to /dev/null
sudo find / -xdev \( -path /proc -o -path /sys -o -path /dev \) -prune -o -type f -printf '%T@ %p\n' 2>/dev/null | sort -n | tail -n 100

or
sudo find / -xdev \( -path /proc -o -path /sys -o -path /dev \) -prune -o -type f -exec stat --format='%Y %n' {} + 2>/dev/null | sort -n | tail -n 100


find the last 100 files that have been created
sudo find / -xdev \( -path /proc -o -path /sys -o -path /dev \) -prune -o -type f -printf '%C@ %p\n' | sort -n | tail -n 100

find the last 100 files that have been accessed (read from)
with unix timestamp 
sudo find / -xdev \( -path /proc -o -path /sys -o -path /dev \) -prune -o -type f -exec stat --format='%X %n' {} + 2>/dev/null | sort -n | tail -n 100

or
sudo find / -xdev \( -path /proc -o -path /sys -o -path /dev \) -prune -o -type f -printf '%A@ %p\n' | sort -n | tail -n 100



note
-xdev  Don't descend directories on other filesystems
